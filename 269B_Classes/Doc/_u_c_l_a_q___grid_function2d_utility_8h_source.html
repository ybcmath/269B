58 void addToValues(GridFunction2d& gFun, std::function<double(double,double)>& F)
59 {
60  long i; long j;
61 
62  double xMin = gFun.getXmin();
63  double yMin = gFun.getYmin();
64 
65  long xPanels = gFun.getXpanelCount();
66  long yPanels = gFun.getYpanelCount();
67 
68  double hx = gFun.getHx();
69  double hy = gFun.getHy();
70 
71  double x; double y;
72 
73  for(i = 0; i <= xPanels; i++)
74  {
75  x = xMin + i*hx;
76  for(j = 0; j <= yPanels; j++)
77  {
78  y = yMin + j*hy;
79  gFun.Values(i,j) += F(x,y);
80  }}

139  void inputFromGNUplot(GridFunction2d& gF, const string& fileName, int& noFileFlag)
140 {
141 
142  //
143  // Open input file
144  //
145  // ifstream dataFile(fileName);
146 
147  FILE* dataFile = 0;
148  noFileFlag = 0;
149  if(OPENFILE(dataFile,fileName.c_str(), "r" ))
150  {
151  noFileFlag = 1;
152  return;
153  }
154 
155  char* strTmp = new char[1024];
156  long mPtmp;
157  long nPtmp;
158 
159  long i; long j;
160  long mPanel; long nPanel;
161  double x; double y;
162 
163  double xMin; double xMax;
164  double yMin; double yMax;
165  double valTmp;
166 
167  if((gF.getXpanelCount() == 0)&&(gF.getYpanelCount() == 0))
168  {
169 
170  // Extract panel count
171 
172  fscanf(dataFile,"%s",strTmp);
173  fscanf(dataFile,"%ld",&mPanel);
174  fscanf(dataFile,"%ld",&nPanel);
175  mPanel -= 1;
176  nPanel -= 1;
177  //
178  // Capture bounds
179  //
180  // x = xMin + double(j)*hx;
181  // y = yMin + double(i)*hy;
182 
183  for(i = 0; i <= nPanel; i++)
184  {
185  for(j = 0; j <= mPanel; j++)
186  {
187  fscanf(dataFile,"%lf %lf %lf",&x,&y,&valTmp);
188  if(j == 0) xMin = x;
189  if(j == nPanel) xMax = x;
190  }
191  if(i == 0) yMin = y;
192  if(i == mPanel) yMax = y;
193  }
194  gF.initialize(mPanel,xMin,xMax,nPanel,yMin,yMax);
195  rewind(dataFile);
196  }
197 
198  UCLAQ::DoubleVector2d* V;
199  V = gF.getValuesPointer();
200 
201  mPanel = gF.getXpanelCount();
202  nPanel = gF.getYpanelCount();
203 
204  fscanf(dataFile,"%s",strTmp);
205  fscanf(dataFile,"%ld",&mPtmp);
206  fscanf(dataFile,"%ld",&nPtmp);
207 
208  for(i = 0; i <= nPanel; i++)
209  {
210  for(j = 0; j <= mPanel; j++)
211  {
212  fscanf(dataFile,"%lf %lf %lf",&x,&y,&(V->operator()(j,i)));
213  }
214  }
215  delete [] strTmp;
216  fclose(dataFile);
217 }
